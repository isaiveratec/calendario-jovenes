    let currentDate = new Date(2026, 1, 1); // Febrero 2026

    let activeFilters = {

    const monthNames = ["Enero", "Febrero", "Marzo", "Abril", "Mayo", "Junio", "Julio", "Agosto", "Septiembre", "Octubre", "Noviembre", "Diciembre"];


    function renderCalendar() {

        const year = currentDate.getFullYear();

        const month = currentDate.getMonth();

        

        document.getElementById('currentMonthDisplay').innerText = `${monthNames[month]} ${year}`;

        

        const firstDayOfMonth = new Date(year, month, 1).getDay();

        const daysInMonth = new Date(year, month + 1, 0).getDate();

        

        const calendarDays = document.getElementById('calendarDays');

        calendarDays.innerHTML = '';


        // Días vacíos mes anterior

        for (let i = 0; i < firstDayOfMonth; i++) {

            const emptyCell = document.createElement('div');

            emptyCell.className = 'day-cell border-r border-b bg-slate-50/30';

            calendarDays.appendChild(emptyCell);

        }


        // Días del mes

        for (let day = 1; day <= daysInMonth; day++) {

            const dateStr = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;

            const dayCell = document.createElement('div');

            dayCell.className = 'day-cell border-r border-b p-3 relative flex flex-col items-start';

            

            const dayNum = document.createElement('span');

            dayNum.className = 'text-sm font-bold text-slate-400 mb-1';

            dayNum.innerText = day;

            dayCell.appendChild(dayNum);


            // Filtro combinado

            const dayEvents = events.filter(e => {

                const dateMatch = e.date === dateStr;

                const groupMatch = activeFilters.group === 'all' || e.group === activeFilters.group;

                const teacherMatch = activeFilters.teacher === 'all' || e.teachers.includes(activeFilters.teacher.split(' ')[0]); // Comparación simplificada por nombre

                

                return dateMatch && groupMatch && teacherMatch;

            });

            

            dayEvents.forEach(event => {

            calendarDays.appendChild(dayCell);

        }

    }


    function changeMonth(delta) {

        currentDate.setMonth(currentDate.getMonth() + delta);

        renderCalendar();

    }


    function setFilter(type, value) {

        activeFilters[type] = value;

        

        // Actualizar UI de botones para el tipo específico

        const container = type === 'group' ? 'filter-group' : 'filter-teacher';

        const buttons = document.querySelectorAll(`button[id^="${container}-"]`);

        

        buttons.forEach(btn => {

            btn.classList.remove('filter-btn-active');

            if (btn.id.includes(value.split(' ')[0])) {

                btn.classList.add('filter-btn-active');

            }

        });


        renderCalendar();

    }


    function openModal(event) {

            modal.classList.remove('active');
